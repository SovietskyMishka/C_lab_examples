#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#define _CRT_SECURE_NO_WARNINGS#include <math.h>
#include <locale.h>	//	Подключение библиотек

void main()
{
	setlocale(LC_ALL, "rus");	//	Подключение кириллицы в консоли
	int n = 0, fact = 1;	//	кол-во шагов и факториал
	double x, EPS, R = 0, L = 0, Sum = 0;	//	значение, погрешность, правая часть, Левая часть, Сумма
	printf("Введите значение:\n");
	scanf("%lf", &x);		//	Ввод значения x
	printf("Введите погрешость:\n");
	scanf("%lf", &EPS);	//	Ввод погрешности
	L = sinh(x);	//	Вычисление гиперболического синуса	- левая часть
	printf("Левая часть = %lf\n", L);
	while (L - R > EPS && n < 100)	//	Цикл вычисления правой части, пока левая часть минус правая больше погрешности
	{
		for (int i = 1; i <= 2 * n + 1; i++) {	//	Цикл вычисления факториала
			fact = fact * i;
		}
		Sum += pow(x, 2 * n + 1) / fact;	//	Суммирование
		fact = 1;	//	Обнуление переменной факториала
		R = Sum;	//	Присваивание правой части
		n++;	//	Прибавление счётчика кол-ва шагов
	}
	printf("Правая часть = %lf\n", R);		//	Вывод результата правой части
	printf("Количество шагов = %d\n", n);	//	Вывод кол-ва шагов
	getchar();	//	Конец программы
	getchar();
}
