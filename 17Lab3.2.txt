#include <stdio.h> /*Ввод и вывод*/
#include <locale.h>	/*для смены на языка*/

void main() {
	setlocale(LC_ALL, "russian");	/*Подключение кириллицы в консоли*/
	int x, y;
	printf("Введите трёхзначное число: ");
	scanf_s("%d", &x);	/*Ввод 3х значного числа*/
	if ((x >= 100) && (x <= 999)) /*Проверка, что x - 3x значное число*/
	{
		while (x > 0)	/*Цикл пока x больше 0*/
		{
			y = x % 2;	/*Остаток от деление на 2*/
			printf("%d", y); /*Вывод в консоль двоичного числа в обратном порядке*/
			x = x / 2;	/*делим x на 2*/
		}
	}
	else /*Если x не 3-х значное число*/
	{
		printf("Введено некорректное значение");
	}
	getchar(); getchar();	/*конец программы*/
}
